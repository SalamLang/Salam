\chapter{متغیرها و انواعِ داده}
\label{ch:variables}

در فصلِ پیش با متغیر آشنا شدیم و آن را به جعبه‌ای تشبیه کردیم که چیزی در آن
نگه می‌داریم. در این فصل ژرف‌تر می‌شویم: انواعِ گوناگونِ داده را می‌شناسیم،
یاد می‌گیریم متغیر را چگونه بسازیم، و درمی‌یابیم چرا «نوعِ» هر متغیر اهمیت
دارد.

\section{متغیر؛ جعبه‌ای با نام}

تصور کنید چند جعبه دارید و روی هر کدام برچسبی زده‌اید: «سن»، «قد»، «نام». در هر
جعبه چیزی گذاشته‌اید. هر وقت به محتوای جعبه نیاز داشتید، کافی است نامِ روی برچسب
را صدا بزنید. در برنامه‌نویسی، این جعبه‌ها همان \textbf{متغیر}ها هستند.

ساده‌ترین شکلِ ساختنِ یک متغیر در سلام چنین است:

\begin{salam}
تابع اصلی:
    متغیر سن : صحیح = 30
    چاپ سن
تمام
\end{salam}

این خط را چنین بخوانید: «یک \textbf{متغیر} به نامِ \textbf{سن} که نوعش
\textbf{صحیح} (عددِ درست) است بساز و مقدارِ \textbf{۳۰} را در آن بگذار.»

ساختار همیشه یکسان است:

\begin{center}
\code{متغیر} \;\textit{نام}\;:\;\textit{نوع} \;=\; \textit{مقدار}
\end{center}

\section{چهار شیوهٔ ساختنِ متغیر}

سلام چهار واژهٔ کلیدی برای ساختنِ متغیر دارد و هر کدام معنایی متفاوت دارند. به
این برنامهٔ کوچک که تقریباً همان نمونهٔ موجود در پوشهٔ مثال‌های سلام است ---
نگاه کنید:

\begin{salam}
تابع اصلی:
    متغیر سن : صحیح = 30
    ثابت سال تولد : صحیح = 1994
    قد : اعشار۶۴ = 1.75
    نام : خودکار = "سارا"
    سن = سن + 1
    چاپ(نام, سن, سال تولد, قد)
تمام
\end{salam}

بیایید چهار شیوه را از هم تفکیک کنیم:

\subsection{متغیر: جعبه‌ای که می‌توان عوضش کرد}

\begin{salam}
متغیر سن : صحیح = 30
\end{salam}

واژهٔ \code{متغیر} (معادلِ \code{mut} در انگلیسی) جعبه‌ای می‌سازد که محتوایش را
بعداً می‌توان \emph{تغییر} داد. به همین دلیل در خطِ \code{سن = سن + 1}
توانستیم مقدارِ \code{سن} را یکی زیاد کنیم؛ حالا \code{سن} برابرِ ۳۱ است.

\subsection{ثابت: جعبه‌ای که قفل است}

\begin{salam}
ثابت سال تولد : صحیح = 1994
\end{salam}

واژهٔ \code{ثابت} (معادلِ \code{const}) جعبه‌ای می‌سازد که پس از پُر شدن،
\emph{دیگر تغییر نمی‌کند}. سالِ تولدِ کسی هیچ‌گاه عوض نمی‌شود، پس درست است که
آن را ثابت بگیریم. اگر بکوشید مقدارِ یک ثابت را تغییر دهید، کامپایلر خطا
می‌گیرد و به شما هشدار می‌دهد.

\subsection{اعلانِ ساده: ثابت اما بی‌واژه}

\begin{salamen}
قد : اعشار۶۴ = 1.75
\end{salamen}

اگر هیچ واژه‌ای (\code{متغیر} یا \code{ثابت}) ننویسید و مستقیم با نام آغاز
کنید، یک متغیرِ \emph{تغییرناپذیر} ساخته می‌شود درست مانندِ \code{ثابت}.
این شکلِ کوتاه و تمیز، شیوهٔ پیش‌فرضِ سلام است: \emph{مگر آنکه واقعاً نیاز به
تغییر داشته باشید، متغیر را تغییرناپذیر بگذارید.}

\subsection{خودکار: بگذارید سلام نوع را حدس بزند}

\begin{salamen}
نام : خودکار = "سارا"
\end{salamen}

گاهی نوشتنِ نوع، تکراری و آزاردهنده است؛ مگر معلوم نیست که \code{"سارا"} یک
متن است؟ واژهٔ \code{خودکار} (معادلِ \code{auto}) به سلام می‌گوید: «خودت از روی
مقدار، نوع را تشخیص بده.» چون \code{"سارا"} میانِ نقل‌قول است، سلام درمی‌یابد
که نوعش \code{رشته} است.

\begin{tip}
استفاده از \code{خودکار} هم کد را کوتاه‌تر می‌کند و هم خواناتر؛ اما در جاهایی
که نوع روشن نیست (مثلاً یک عددِ خام که نمی‌دانید درست است یا اعشاری)، نوشتنِ
صریحِ نوع به خوانندهٔ کدِ شما کمک می‌کند.
\end{tip}

\begin{note}
به دو شیوهٔ صدا زدنِ \code{چاپ} دقت کنید: در فصلِ پیش نوشتیم
\code{چاپ "سلام،", نام} (بدونِ پرانتز) و اینجا نوشتیم
\code{چاپ(نام, سن, ...)} (با پرانتز). هر دو درست‌اند. پرانتز اختیاری است، اما
بسیاری از برنامه‌نویسان آن را برای روشنیِ بیشتر می‌پسندند. در این کتاب هر دو
شکل را خواهید دید.
\end{note}

\section{انواعِ پایهٔ داده}

هر متغیر یک \textbf{نوع} دارد که می‌گوید چه جور داده‌ای را می‌تواند نگه دارد.
نوع، هم به سلام کمک می‌کند که برنامهٔ شما را درست بفهمد، و هم از بسیاری خطاها
جلوگیری می‌کند. مهم‌ترین انواعِ پایه در سلام این‌ها هستند:

\subsection{اعدادِ درست (صحیح)}

عددهای بدونِ اعشار مانندِ ۰، ۷، ۱۵، و ۱۰۰- را اعدادِ \emph{درست} یا
\emph{صحیح} می‌نامیم. سلام چند نوعِ عددِ درست دارد که در \emph{اندازه} با هم
فرق می‌کنند:

\begin{center}
\begin{tabular}{c c l}
\toprule
\textbf{نامِ فارسی} & \textbf{نامِ انگلیسی} & \textbf{توضیح} \\
\midrule
\code{صحیح} & \code{int} / \code{i32} & عددِ درستِ معمول (۳۲ بیتی) \\
\code{صحیح۳۲} & \code{i32} & عددِ درستِ ۳۲ بیتی \\
\code{صحیح۶۴} & \code{i64} & عددِ درستِ ۶۴ بیتی (برای عددهای خیلی بزرگ) \\
\bottomrule
\end{tabular}
\end{center}

برای کارهای روزمره \code{صحیح} کافی است. اگر با عددهای بسیار بزرگ سر و کار
دارید (مثلاً فاکتوریلِ اعداد یا شمارشِ چیزهای فراوان)، از \code{صحیح۶۴}
استفاده کنید.

\begin{note}
\code{صحیح} و \code{صحیح۳۲} دقیقاً \emph{یک} نوع‌اند: هر دو همان عددِ درستِ ۳۲
بیتی (\code{i32}) هستند و هر جا یکی را بنویسید، دیگری هم همان معنا را می‌دهد.
عددِ درست در سلام به‌طورِ پیش‌فرض ۳۲ بیتی است، پس نوشتنِ \code{۳۲} در پایانِ
نام ضروری نیست. در سراسرِ این کتاب همان شکلِ کوتاه، یعنی \code{صحیح}، را به‌کار
می‌بریم؛ کوتاه‌تر است و کد را خواناتر نگه می‌دارد. \code{صحیح۳۲} را تنها وقتی
بنویسید که بخواهید بر پهنای ۳۲ بیتیِ عدد تأکید کنید.
\end{note}

\begin{salam}
تابع اصلی:
    متغیر کوچک : صحیح = 1000
    متغیر بزرگ : صحیح۶۴ = 9000000000
    چاپ(کوچک, بزرگ)
تمام
\end{salam}

\subsection{اعدادِ اعشاری}

عددهایی که بخشِ اعشاری دارند مانندِ ۱٫۷۵، ۳٫۱۴ و ۰٫۵ را اعدادِ
\emph{اعشاری} می‌نامیم. در سلام نوعِ آن‌ها \code{اعشار۶۴} است (معادلِ
\code{f64}):

\begin{salam}
تابع اصلی:
    قد : اعشار۶۴ = 1.75
    وزن : اعشار۶۴ = 68.5
    چاپ(قد, وزن)
تمام
\end{salam}

\begin{warn}
در عددهای اعشاری، نقطهٔ اعشار همیشه با نقطهٔ لاتین (\code{.}) نوشته می‌شود، نه
ممیزِ فارسی. پس بنویسید \code{1.75} نه «۱٫۷۵». همچنین برای آنکه عددی اعشاری
شمرده شود، باید نقطه داشته باشد: \code{2} یک عددِ درست است اما \code{2.0} یک
عددِ اعشاری.
\end{warn}

\subsection{متن (رشته)}

هر دنباله‌ای از نویسه‌ها یک واژه، یک جمله، یا حتی یک نام یک
\emph{رشته} (string) است. رشته‌ها را همیشه میانِ دو نقل‌قول می‌نویسیم و نوعشان
\code{رشته} است (معادلِ \code{str}):

\begin{salam}
تابع اصلی:
    شهر : رشته = "اصفهان"
    پیام : رشته = "به سلام خوش آمدید"
    چاپ(شهر, پیام)
تمام
\end{salam}

\subsection{مقادیرِ منطقی (درست/نادرست)}

گاهی فقط دو پاسخ ممکن است: آری یا نه، روشن یا خاموش، درست یا نادرست. برای این
موارد نوعِ \code{منطقی} (معادلِ \code{bool}) را داریم که تنها دو مقدار می‌پذیرد:
\code{درست} (true) و \code{نادرست} (false):

\begin{salam}
تابع اصلی:
    باران : منطقی = درست
    آفتاب : منطقی = نادرست
    چاپ(باران, آفتاب)
تمام
\end{salam}

مقادیرِ منطقی هنگامِ تصمیم‌گیری بسیار به‌کار می‌آیند؛ در فصلِ شرط‌ها (فصلِ
\ref{ch:conditionals}) نقشِ کلیدیِ آن‌ها را خواهیم دید.

\subsection{کاراکتر (تک‌نویسه)}

اگر تنها به \emph{یک} نویسه نیاز داشته باشید مثلاً حرفِ \lr{A} یا رقمِ ۷ ---
از نوعِ \code{کاراکتر} (معادلِ \code{char}) استفاده می‌کنید. کاراکترها را میانِ
\emph{تک‌نقل‌قول} می‌نویسیم:

\begin{salam}
تابع اصلی:
    حرف : کاراکتر = 'A'
    چاپ حرف
تمام
\end{salam}

\begin{note}
تفاوتِ \code{'A'} و \code{"A"} چیست؟ نخستی یک \emph{کاراکتر} (تک‌نویسه) است و
دومی یک \emph{رشته} (دنباله‌ای از نویسه‌ها که اینجا فقط یک نویسه دارد). در
بسیاری از زبان‌ها این تمایز مهم است.
\end{note}

\section{جدولِ کاملِ انواعِ پایه}

\begin{center}
\begin{tabular}{c c l}
\toprule
\textbf{فارسی} & \textbf{انگلیسی} & \textbf{چه چیزی نگه می‌دارد} \\
\midrule
\code{صحیح} / \code{صحیح۳۲} & \code{int} / \code{i32} & عددِ درستِ ۳۲ بیتی \\
\code{صحیح۶۴} & \code{i64} & عددِ درستِ ۶۴ بیتی \\
\code{اعشار۶۴} & \code{f64} & عددِ اعشاری \\
\code{اعشار۳۲} & \code{f32} & عددِ اعشاریِ کوچک‌تر \\
\code{منطقی} & \code{bool} & درست یا نادرست \\
\code{کاراکتر} & \code{char} & یک نویسه \\
\code{رشته} & \code{str} & دنباله‌ای از نویسه‌ها \\
\code{خودکار} & \code{auto} & نوع را خودِ سلام تشخیص می‌دهد \\
\bottomrule
\end{tabular}
\end{center}

\begin{tip}
هر پرونده در سلام به \emph{یک} زبان نوشته می‌شود و نام‌های نوع هم باید به همان
زبان باشند. در یک برنامهٔ فارسی باید بنویسید \code{متغیر سن: صحیح = 30}؛ اگر
نامِ انگلیسیِ همان نوع، یعنی \code{i32}، را در پروندهٔ فارسی به‌کار ببرید، کامپایلر
خطا می‌گیرد و می‌گوید نامِ نوع در یک پروندهٔ فارسی باید فارسی باشد. نامِ انگلیسیِ
انواع تنها در پرونده‌های انگلیسی معتبر است.
\end{tip}

\section{تبدیلِ نوع با «بعنوان»}

گاهی می‌خواهید یک مقدار را از نوعی به نوعِ دیگر تبدیل کنید؛ مثلاً یک عددِ درست
را به اعشاری، یا برعکس. برای این کار از واژهٔ \code{بعنوان} (معادلِ \code{as})
استفاده می‌کنید:

\begin{salam}
تابع اصلی:
    متغیر شمار : صحیح = 7
    متغیر دقیق : اعشار۶۴ = شمار بعنوان اعشار۶۴
    چاپ(دقیق)
تمام
\end{salam}

اینجا \code{شمار} (که یک عددِ درست است) به \code{اعشار۶۴} تبدیل شد. به این کار
\textbf{تبدیلِ نوع} (cast) می‌گویند و در فصل‌های بعد به‌ویژه هنگامِ کار با
شمارشی‌ها و آرایه‌ها بارها به آن نیاز خواهیم داشت.

\section{قاعده‌های نام‌گذاری}

نامِ متغیرها را خودتان برمی‌گزینید، اما چند قاعدهٔ ساده هست:

\begin{itemize}
  \item نام می‌تواند فارسی، انگلیسی یا ترکیبی باشد: \code{سن}، \code{age}،
        \code{شمارهٔ‌حساب}.
  \item نام می‌تواند چندواژه‌ای باشد؛ بینِ واژه‌ها فاصله بگذارید:
        \code{سال تولد}، \code{شمارنده اصلی}. (زیرخط \code{\_} هم مجاز است.)
  \item نام نمی‌تواند با یک رقم آغاز شود.
  \item نام نباید یکی از واژگانِ کلیدیِ زبان (مانندِ \code{تابع} یا
        \code{اگر}) باشد.
\end{itemize}

\begin{tip}
نامِ خوب، نامی است که \emph{معنا}ی متغیر را برساند. \code{مجموعِ‌نمرات} بسیار
گویاتر از \code{م} است. کدی که شش ماه بعد بخوانید، باید همچنان قابلِ فهم باشد ---
و نامِ خوب نیمی از این فهم را تأمین می‌کند.
\end{tip}

\begin{deep}[نگاهی به درون: نوع چگونه به ایمنی کمک می‌کند]
سلام یک زبانِ \emph{ایستا-نوع} (statically typed) است؛ یعنی نوعِ هر متغیر پیش
از اجرا و در زمانِ کامپایل مشخص است. این ویژگی به کامپایلر اجازه می‌دهد بسیاری
از خطاها را \emph{پیش} از اجرای برنامه بگیرد. برای نمونه، اگر بکوشید یک رشته را
با یک عدد جمع بزنید، یا مقدارِ اعشاری را در جعبه‌ای از نوعِ صحیح بگذارید،
کامپایلر همان لحظه به شما هشدار می‌دهد به‌جای آنکه برنامه هنگامِ اجرا و
شاید در دستِ کاربر، ناگهان از کار بیفتد.
\end{deep}

\section{خلاصهٔ فصل}

\begin{itemize}
  \item متغیر، جعبه‌ای نام‌دار برای نگه‌داشتنِ داده است.
  \item چهار شیوهٔ اعلان داریم: \code{متغیر} (تغییرپذیر)، \code{ثابت}
        (تغییرناپذیر)، اعلانِ ساده (تغییرناپذیر)، و \code{خودکار} (تشخیصِ
        خودکارِ نوع).
  \item انواعِ پایه: \code{صحیح}، \code{صحیح۶۴}، \code{اعشار۶۴}،
        \code{منطقی}، \code{کاراکتر} و \code{رشته}.
  \item با \code{بعنوان} می‌توان نوعِ یک مقدار را تبدیل کرد.
  \item نام‌های فارسی و انگلیسیِ انواع هر دو معتبرند.
\end{itemize}

\section{تمرین‌ها}

\exercise سه متغیر بسازید: نامِ شهر (رشته)، جمعیتِ آن (صحیح۶۴) و میانگینِ دمای
سالانه (اعشار۶۴). سپس هر سه را در یک خط چاپ کنید.

\exercise یک ثابت به نامِ \code{پی} با مقدارِ \code{3.14} بسازید و کوشش کنید
در خطِ بعد مقدارش را تغییر دهید. کامپایلر چه می‌گوید؟

\exercise متغیری از نوعِ \code{منطقی} بسازید که نشان دهد امروز تعطیل است یا
نه، و آن را چاپ کنید.

\exercise عددِ درستِ \code{9} را در یک متغیر بگذارید، سپس آن را \code{بعنوان
اعشار۶۴} تبدیل کرده و در متغیرِ دیگری بگذارید و هر دو را چاپ کنید.

\exercise برنامه‌ای بنویسید که از \code{خودکار} برای ساختنِ سه متغیر (یک عدد،
یک اعشاری و یک رشته) استفاده کند و هر سه را چاپ کند.
