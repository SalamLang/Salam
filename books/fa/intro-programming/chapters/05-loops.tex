\chapter{حلقه‌ها}
\label{ch:loops}

رایانه در یک کار بسیار توانمند است: \emph{تکرار}. کارهایی که برای انسان
خسته‌کننده و طولانی‌اند مثلاً جمع‌زدنِ هزار عدد یا چاپِ صد سطر را رایانه
در یک چشم‌به‌هم‌زدن انجام می‌دهد. ابزارِ این تکرار، \textbf{حلقه} است. در این
فصل با همهٔ حلقه‌های سلام آشنا می‌شویم.

\section{تاوقتی: تا زمانی که شرط برقرار است}

ساده‌ترین حلقه، \code{تاوقتی} (معادلِ \code{while}) است. این حلقه بدنهٔ خود را
\emph{تا زمانی که} یک شرط درست است، بارها و بارها اجرا می‌کند. به این مثالِ
واقعی نگاه کنید که اعدادِ ۱ تا ۱۰ را جمع می‌زند:

\begin{salam}
تابع اصلی:
    متغیر مجموع صحیح۳۲ = 0
    متغیر i صحیح۳۲ = 1
    تاوقتی i <= 10:
        مجموع = مجموع + i
        i = i + 1
    پایان
    چاپ("مجموع ۱ تا ۱۰ برابر است با", مجموع)
پایان
\end{salam}

این حلقه را گام‌به‌گام دنبال کنیم:
\begin{enumerate}
  \item نخست \code{مجموع} برابرِ ۰ و \code{i} برابرِ ۱ است.
  \item سلام شرطِ \code{i <= 10} را می‌آزماید. درست است، پس بدنه اجرا می‌شود:
        \code{مجموع} برابرِ ۱ و \code{i} برابرِ ۲ می‌شود.
  \item دوباره شرط آزموده می‌شود. باز درست است... و این کار تا وقتی \code{i}
        به ۱۱ برسد ادامه می‌یابد.
  \item وقتی \code{i} برابرِ ۱۱ شد، شرط \emph{نادرست} می‌شود و حلقه پایان
        می‌یابد.
\end{enumerate}

خروجی: \code{مجموع ۱ تا ۱۰ برابر است با 55}.

\begin{warn}
هر حلقهٔ \code{تاوقتی} به دو چیز نیاز دارد: شرطی که سرانجام نادرست شود، و
تغییری در بدنه که ما را به آن نزدیک کند. اگر خطِ \code{i = i + 1} را فراموش
کنید، \code{i} همیشه ۱ می‌ماند، شرط همیشه درست است و حلقه \emph{هرگز پایان
نمی‌یابد}! به این اشتباهِ رایج، \textbf{حلقهٔ بی‌پایان} می‌گویند.
\end{warn}

\section{برای: حلقهٔ شمارنده‌دار}

بسیار پیش می‌آید که می‌خواهیم کاری را به تعدادِ مشخصی تکرار کنیم و در ضمن یک
شمارنده هم داشته باشیم. حلقهٔ \code{برای} (معادلِ \code{for}) دقیقاً برای همین
ساخته شده و هر سه بخشِ یک شمارنده را در یک خط گرد می‌آورد:

\begin{salam}
تابع اصلی:
    برای متغیر i = 1; i <= 10; i = i + 1:
        چاپ(5, "*", i, "=", 5 * i)
    پایان
پایان
\end{salam}

خطِ \code{برای} سه بخش دارد که با نقطه‌ویرگول (\code{;}) از هم جدا می‌شوند:

\begin{center}
\code{برای} \;\textit{آغاز}\; \code{;} \;\textit{شرط}\; \code{;} \;\textit{گام}\; \code{:}
\end{center}

\begin{itemize}
  \item \textbf{آغاز} (\code{متغیر i = 1}): یک‌بار و در ابتدا اجرا می‌شود؛
        شمارنده را می‌سازد.
  \item \textbf{شرط} (\code{i <= 10}): پیش از هر تکرار آزموده می‌شود؛ تا درست
        است، حلقه ادامه دارد.
  \item \textbf{گام} (\code{i = i + 1}): پس از هر تکرار اجرا می‌شود؛ معمولاً
        شمارنده را پیش می‌برد.
\end{itemize}

این برنامه جدولِ ضربِ ۵ را چاپ می‌کند. در عمل، حلقهٔ \code{برای} و \code{تاوقتی}
هم‌ارزند؛ هر کدام را با دیگری می‌توان نوشت. اما وقتی شمارنده‌ای روشن دارید،
\code{برای} خواناتر و جمع‌وجورتر است.

\begin{tip}
نامِ \code{i} برای شمارندهٔ حلقه یک سنّتِ دیرینه است (کوتاه‌شدهٔ \emph{index}).
برای حلقه‌های تودرتو از \code{i}، \code{j} و \code{k} استفاده می‌کنند. البته
می‌توانید نام‌های فارسیِ گویاتر مانندِ \code{ردیف} و \code{ستون} هم به‌کار
ببرید.
\end{tip}

\section{شکل‌کشی با حلقه‌های تودرتو}

وقتی یک حلقه را \emph{درونِ} حلقهٔ دیگر بگذاریم، کارهای جالبی می‌توان کرد. این
مثالِ واقعی یک مثلثِ ستاره می‌کشد نمونه‌ای محبوب نزدِ تازه‌کارها:

\begin{salam}
// مثلث ستاره
تابع اصلی:
    برای متغیر ردیف = 1; ردیف <= 5; ردیف = ردیف + 1:
        متغیر خط رشته = ""
        برای متغیر ستون = 0; ستون < ردیف; ستون = ستون + 1: خط = خط + "*" پایان
        چاپ خط
    پایان
پایان
\end{salam}

خروجی:

\begin{salamen}[language={}]
*
**
***
****
*****
\end{salamen}

حلقهٔ بیرونی روی ردیف‌ها (۱ تا ۵) می‌چرخد، و برای هر ردیف، حلقهٔ درونی به تعدادِ
شمارهٔ آن ردیف ستاره به رشتهٔ \code{خط} می‌افزاید. سپس خط چاپ می‌شود. این الگوی
«ساختنِ یک رشته در حلقهٔ درونی و چاپِ آن در حلقهٔ بیرونی» را بارها به‌کار خواهید
برد.

\begin{tip}
همین برنامه را می‌توان با جدولِ ضرب هم نوشت: کافی است به‌جای افزودنِ ستاره،
حاصلِ \code{ردیف * ستون} را به خط بیفزایید. مثال‌های واقعیِ سلام چنین برنامه‌ای
نیز دارند.
\end{tip}

\section{تکرار: ساده‌ترین راهِ تکرار}

گاهی فقط می‌خواهیم کاری را \emph{چند بار} انجام دهیم و به شمارنده کاری نداریم.
برای این، سلام واژهٔ زیبای \code{تکرار} (معادلِ \code{repeat}) را دارد که سه
شکل دارد. این مثالِ واقعی هر سه را نشان می‌دهد:

\begin{salam}
تابع اصلی:
    // ۱) شکل شمارشی: بدنه را N بار اجرا می‌کند
    تکرار 5:
        چاپ "تست"
    پایان

    // شمارنده می‌تواند هر عبارت عددی باشد (یک‌بار محاسبه می‌شود)
    تعداد صحیح = 3
    تکرار تعداد:
        بنویس "x"
    پایان
    چاپ ""

    // ۲) شکل بازه‌ای: از یک آغاز تا یک پایانِ شامل
    تکرار 1 تا 5:
        بنویس "#"
    پایان
    چاپ ""

    // ۳) شکل بازه‌ای با گام: ۰،۱۰،۲۰،...،۱۰۰
    تکرار 0 تا 100 هر 10:
        بنویس "."
    پایان
    چاپ ""
پایان
\end{salam}

سه شکلِ \code{تکرار} چنین‌اند:

\begin{enumerate}
  \item \textbf{شمارشی} (\code{تکرار N:}): بدنه را \code{N} بار اجرا می‌کند.
  \item \textbf{بازه‌ای} (\code{تکرار آ تا ب:}): برای هر مقدار از \code{آ} تا
        \code{ب} (شاملِ خودِ \code{ب}) یک‌بار اجرا می‌شود.
  \item \textbf{بازه‌ای با گام} (\code{تکرار آ تا ب هر گ:}): از \code{آ} تا
        \code{ب} با گام‌های \code{گ} پیش می‌رود.
\end{enumerate}

\begin{note}
\code{تکرار} شمارندهٔ آشکاری در اختیارِ شما نمی‌گذارد یعنی درونِ بدنه به
شمارهٔ تکرارِ جاری دسترسی ندارید. اگر به آن نیاز دارید، یا یک شمارندهٔ دستی نگه
دارید (مثلِ مثالِ زیر) یا از حلقهٔ \code{برای} استفاده کنید.
\end{note}

این مثالِ واقعی نشان می‌دهد چگونه با نگه‌داشتنِ یک شمارندهٔ دستی، با
\code{تکرار} هم می‌توان جمع زد و هم شکل کشید:

\begin{salam}
// «تکرار آ تا ب:» بدنه را برای هر مقدار از آ تا بِ (شامل) یک‌بار اجرا می‌کند.
تابع اصلی:
    // جمع ۱ تا ۱۰
    متغیر مجموع صحیح = 0
    متغیر ک صحیح = 0
    تکرار 1 تا 10:
        ک = ک + 1
        مجموع = مجموع + ک
    پایان
    بنویس "۱+۲+...+۱۰ = "
    چاپ مجموع

    // مثلثِ چپ‌چین: سطر r تعداد r ستاره چاپ می‌کند (سطرهای ۱ تا ۵)
    متغیر سطر صحیح = 0
    تکرار 1 تا 5:
        سطر = سطر + 1
        تکرار 1 تا سطر:
            بنویس "*"
        پایان
        چاپ ""
    پایان
پایان
\end{salam}

\section{بشکن و ادامه: کنترلِ دقیق‌ترِ حلقه}

گاهی می‌خواهیم پیش از پایانِ طبیعیِ حلقه از آن بیرون بیاییم، یا یک تکرار را رد
کنیم. برای این دو کار، \code{بشکن} (معادلِ \code{break}) و \code{ادامه}
(معادلِ \code{continue}) را داریم. به این مثالِ واقعی نگاه کنید:

\begin{salam}
تابع اصلی:
    متغیر مجموع i32 = 0
    متغیر i i32 = 0
    تاوقتی i < 20:
        i = i + 1
        اگر i % 2 == 0:
            ادامه
        پایان
        اگر i > 9:
            بشکن
        پایان
        مجموع = مجموع + i
    پایان
    چاپ مجموع
پایان
\end{salam}

در این حلقه:
\begin{itemize}
  \item \code{ادامه} وقتی \code{i} زوج است اجرا می‌شود و باعث می‌شود سلام
        \emph{باقیِ} بدنه را رد کند و به تکرارِ بعد برود. پس اعدادِ زوج هرگز به
        مجموع افزوده نمی‌شوند.
  \item \code{بشکن} وقتی \code{i} از ۹ بزرگ‌تر شود، حلقه را \emph{به‌کلی}
        پایان می‌دهد.
\end{itemize}
در نتیجه فقط اعدادِ فردِ ۱ تا ۹ (یعنی ۱، ۳، ۵، ۷، ۹) جمع می‌شوند و خروجی
\code{25} می‌شود.

\begin{tip}
\code{بشکن} برای زمانی عالی است که دنبالِ چیزی می‌گردید و همین که یافتیدش، دیگر
نیازی به ادامهٔ جست‌وجو نیست. \code{ادامه} هم برای رد کردنِ موارد ناخواسته
(مثلِ اعدادِ زوج در مثالِ بالا) به‌کار می‌آید.
\end{tip}

\section{مثالِ کامل: بازیِ فیزباز}

«فیزباز» تمرینی کلاسیک است که حلقه، شرط و باقی‌مانده را با هم به کار می‌گیرد.
قاعده ساده است: از ۱ تا ۱۵ بشمار؛ به‌جای مضرب‌های ۳ بنویس «فیز»، به‌جای
مضرب‌های ۵ بنویس «باز»، و به‌جای مضرب‌های هر دو بنویس «فیزباز». این مثالِ واقعیِ
سلام است:

\begin{salam}
// فیزباز
تابع اصلی:
    برای متغیر i = 1; i <= 15; i = i + 1:
        اگر i % 15 == 0:      چاپ "فیزباز"
        وگرنه اگر i % 3 == 0: چاپ "فیز"
        وگرنه اگر i % 5 == 0: چاپ "باز"
        وگرنه:               چاپ i
        پایان
    پایان
پایان
\end{salam}

\begin{warn}
چرا نخست \code{i \% 15 == 0} را می‌آزماییم؟ چون عددی که هم بر ۳ و هم بر ۵
بخش‌پذیر است، حتماً بر ۱۵ هم بخش‌پذیر است. اگر این شرط را \emph{آخر}
می‌گذاشتیم، عددِ ۱۵ زودتر در شرطِ \code{\% 3} گیر می‌افتاد و به‌اشتباه فقط
«فیز» چاپ می‌شد. باز هم می‌بینیم که \emph{ترتیبِ شرط‌ها} سرنوشت‌ساز است.
\end{warn}

\section{خلاصهٔ فصل}

\begin{itemize}
  \item \code{تاوقتی} تا زمانی که شرط درست است تکرار می‌کند.
  \item \code{برای} حلقه‌ای با آغاز، شرط و گام در یک خط است.
  \item حلقه‌های تودرتو برای شکل‌کشی و جدول‌ها عالی‌اند.
  \item \code{تکرار} ساده‌ترین حلقه است و سه شکل دارد: شمارشی، بازه‌ای و
        بازه‌ای با گام.
  \item \code{بشکن} حلقه را پایان می‌دهد و \code{ادامه} به تکرارِ بعد می‌پرد.
  \item مراقبِ حلقه‌های بی‌پایان باشید!
\end{itemize}

\section{تمرین‌ها}

\exercise با حلقهٔ \code{برای}، اعدادِ ۱ تا ۲۰ را چاپ کنید.

\exercise با حلقهٔ \code{تاوقتی}، جمعِ اعدادِ زوجِ ۱ تا ۱۰۰ را حساب کنید.

\exercise با \code{تکرار} و یک شمارندهٔ دستی، فاکتوریلِ عددِ ۵ را حساب کنید
(یعنی $1\times2\times3\times4\times5$).

\exercise یک مثلثِ \emph{راست‌چین} با ۵ ردیف بکشید (هر ردیف چند فاصله و سپس
ستاره دارد).

\exercise جدولِ ضربِ کاملِ ۱ تا ۵ را با دو حلقهٔ تودرتو چاپ کنید.

\exercise با \code{بشکن}، برنامه‌ای بنویسید که اعداد را از ۱ بشمارد و به محضِ
رسیدن به نخستین عددی که بر ۷ بخش‌پذیر است، آن را چاپ کند و بایستد.
