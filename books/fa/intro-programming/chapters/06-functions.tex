\chapter{توابع}
\label{ch:functions}

تا اینجا چند بار با \code{تابع} روبه‌رو شده‌ایم دستِ‌کم تابعِ \code{اصلی}
را در هر برنامه دیده‌ایم. در این فصل توابع را به‌طورِ کامل می‌شناسیم: چرا به
آن‌ها نیاز داریم، چگونه می‌سازیمشان، چگونه به آن‌ها ورودی می‌دهیم و چگونه از
آن‌ها خروجی می‌گیریم.

\section{تابع چیست و چرا به آن نیاز داریم؟}

تابع، بسته‌ای از دستورهاست که نامی دارد و کاری مشخص انجام می‌دهد. تصور کنید
دستورِ پختِ یک کیک را یک‌بار نوشته‌اید؛ از آن پس هر وقت کیک خواستید، کافی است
بگویید «کیک بپز» لازم نیست هر بار همهٔ مراحل را تکرار کنید. تابع همین نقش
را دارد: یک‌بار می‌نویسیدش و بارها \emph{صدایش می‌زنید}.

توابع سه فایدهٔ بزرگ دارند:
\begin{itemize}
  \item \textbf{پرهیز از تکرار:} کاری که چند بار لازم می‌شود را یک‌بار
        می‌نویسید.
  \item \textbf{خوانایی:} برنامه به بخش‌های کوچک و معنادار تقسیم می‌شود.
  \item \textbf{آزمون‌پذیری:} هر تابع را جداگانه می‌توان آزمود و درست کرد.
\end{itemize}

\section{ساختنِ نخستین تابع}

به این مثالِ واقعی نگاه کنید که دو تابعِ کوچک می‌سازد:

\begin{salam}
تابع توان(پایه صحیح۳۲, نما صحیح۳۲) صحیح۳۲:
    متغیر نتیجه صحیح۳۲ = 1
    برای متغیر i = 0; i < نما; i = i + 1:
        نتیجه = نتیجه * پایه
    پایان
    بازگشت نتیجه
پایان

تابع بیشینه(الف صحیح۳۲, ب صحیح۳۲) صحیح۳۲:
    اگر الف > ب:
        بازگشت الف
    پایان
    بازگشت ب
پایان

تابع اصلی:
    چاپ("دو به توان ده:", توان(2, 10))
    چاپ("بیشینه:", بیشینه(توان(2, 4), توان(3, 2)))
پایان
\end{salam}

ساختارِ کلیِ یک تابع چنین است:

\begin{center}
\code{تابع} \;\textit{نام}\;\code{(}\textit{ورودی‌ها}\code{)}\; \textit{نوعِ خروجی}\; \code{:} \quad\ldots\;بدنه\;\ldots\quad \code{پایان}
\end{center}

بیایید تابعِ \code{توان} را بشکافیم:
\begin{itemize}
  \item \code{توان} نامِ تابع است.
  \item \code{(پایه صحیح۳۲, نما صحیح۳۲)} دو \textbf{پارامتر} (ورودی) است: هر
        کدام نامی و نوعی دارند و با کاما جدا شده‌اند.
  \item \code{صحیح۳۲} پس از پرانتز، \textbf{نوعِ خروجیِ} تابع است؛ یعنی این
        تابع یک عددِ صحیح برمی‌گرداند.
  \item درونِ بدنه، توان را با یک حلقه حساب می‌کنیم و با \code{بازگشت نتیجه}
        آن را بیرون می‌دهیم.
\end{itemize}

\section{پارامترها: ورودیِ تابع}

پارامترها مانندِ متغیرهایی هستند که هنگامِ \emph{صدا زدنِ} تابع، مقدار
می‌گیرند. وقتی می‌نویسیم \code{توان(2, 10)}، عددِ \code{2} در \code{پایه} و
عددِ \code{10} در \code{نما} می‌نشیند. ترتیب اهمیت دارد: نخستین مقدار به نخستین
پارامتر می‌رود و الی آخر.

\begin{note}
هر تابع می‌تواند هیچ، یک، یا چند پارامتر داشته باشد. تابعِ \code{اصلی} معمولاً
بی‌پارامتر است؛ به همین دلیل پس از نامش پرانتزِ خالی نمی‌گذاریم و مستقیم
\code{تابع اصلی:} می‌نویسیم.
\end{note}

\section{مقدارهای پیش‌فرضِ پارامتر}

یک پارامتر می‌تواند با \code{=} یک \textbf{مقدارِ پیش‌فرض} داشته باشد. هرگاه
فراخواننده آن آرگومان را ننویسد، مقدارِ پیش‌فرض به‌طورِ خودکار جایش می‌نشیند. به
این ترتیب یک تابع هم حالتِ پرکاربرد را پوشش می‌دهد و هم حالتِ استثنا را،
بی‌آنکه به نسخهٔ جداگانه‌ای از تابع نیاز باشد:

\begin{salam}
تابع درود(نام رشته, پیشوند رشته = "سلام", بار صحیح۳۲ = 1):
    تکرار بار:
        چاپ(پیشوند, نام)
    پایان
پایان

تابع اصلی:
    درود("سارا")              // پیشوند = "سلام"، بار = ۱
    درود("علی", "های")        // بار = ۱
    درود("مکس", "هی", 3)      // همه داده شده
پایان
\end{salam}

\begin{progout}
سلام سارا
های علی
هی مکس
هی مکس
هی مکس
\end{progout}

عبارتِ پیش‌فرض \emph{هنگامِ فراخوانی} و هر بار که آرگومان نوشته نشود ارزیابی
می‌شود؛ پس می‌تواند به ثابت‌ها و دیگر نام‌های سطح‌بالا اشاره کند، اما نه به
پارامترهای دیگرِ همان تابع:

\begin{salam}
ثابت گام صحیح۳۲ = 5
تابع افزایش(عدد صحیح۳۲, مقدار صحیح۳۲ = گام) صحیح۳۲:
    بازگشت عدد + مقدار
پایان

تابع اصلی:
    چاپ(افزایش(10))       // مقدار = گام = ۵
    چاپ(افزایش(10, 1))    // مقدار = ۱
پایان
\end{salam}

\begin{progout}
15
11
\end{progout}

\begin{warn}
پیش‌فرض‌ها باید در \emph{انتها} باشند: پس از نخستین پارامترِ دارای پیش‌فرض، همهٔ
پارامترهای بعدی هم باید پیش‌فرض داشته باشند. پس
\code{(الف صحیح۳۲, ب صحیح۳۲ = 0)} درست است، اما
\code{(الف صحیح۳۲ = 0, ب صحیح۳۲)} پذیرفته نمی‌شود اگر تنها یک آرگومان بدهید،
کامپایلر نمی‌تواند تشخیص دهد کدام پارامتر را در نظر داشته‌اید. مقدارِ پیش‌فرض هم
باید با نوعِ پارامتر سازگار باشد، پس \code{مقدار صحیح۳۲ = "پنج"} خطای زمانِ
کامپایل است.
\end{warn}

\section{بازگشت: خروجیِ تابع}

واژهٔ \code{بازگشت} (معادلِ \code{ret}) دو کار هم‌زمان انجام می‌دهد: مقداری را
به بیرونِ تابع می‌فرستد، و بی‌درنگ از تابع خارج می‌شود. در تابعِ \code{بیشینه}،
اگر \code{الف > ب} باشد، \code{بازگشت الف} اجرا می‌شود و باقیِ تابع هرگز
اجرا نمی‌شود.

\begin{tip}
از همین ویژگیِ «خروجِ بی‌درنگ» می‌توان برای ساده‌تر کردنِ کد استفاده کرد.
تابعِ \code{بیشینه} را ببینید: به‌جای نوشتنِ \code{اگر...وگرنه}، نخست حالتِ
خاص را با \code{بازگشت} مدیریت می‌کنیم و باقی را در ادامه می‌نویسیم. این الگو
به «بازگشتِ زودهنگام» معروف است و کد را تخت‌تر و خواناتر می‌کند.
\end{tip}

\subsection{توابعِ بدونِ خروجی}

هر تابعی لازم نیست چیزی برگرداند. تابعی که فقط کاری انجام می‌دهد (مثلاً چیزی
چاپ می‌کند) نوعِ خروجی ندارد:

\begin{salam}
تابع سلام_بگو(نام رشته):
    چاپ("سلام،", نام)
    چاپ("به سلام خوش آمدی!")
پایان

تابع اصلی:
    سلام_بگو("سارا")
    سلام_بگو("علی")
پایان
\end{salam}

دقت کنید که پس از \code{(نام رشته)} هیچ نوعِ خروجی نیامده چون این تابع چیزی
برنمی‌گرداند. در این‌گونه توابع، \code{بازگشت} (بدونِ مقدار) را می‌توان برای
خروجِ زودهنگام به‌کار برد، اما اجباری نیست.

\section{بازگشتی: تابعی که خودش را صدا می‌زند}

یکی از زیباترین ایده‌های برنامه‌نویسی این است که یک تابع می‌تواند \emph{خودش}
را صدا بزند. به این کار \textbf{بازگشت} یا \textbf{بازگشتی} (recursion)
می‌گویند. مثالِ کلاسیک، فاکتوریل است:

\begin{salam}
تابع فاکتوریل(ن i32) i32:
    اگر ن <= 1:
        بازگشت 1
    پایان
    بازگشت ن * فاکتوریل(ن - 1)
پایان

تابع اصلی:
    چاپ فاکتوریل(5)
    چاپ فاکتوریل(7)
پایان
\end{salam}

فاکتوریلِ ۵ یعنی $5\times4\times3\times2\times1 = 120$. تابع چنین می‌اندیشد:
«فاکتوریلِ ۵ برابر است با ۵ ضربدرِ فاکتوریلِ ۴، و فاکتوریلِ ۴ برابر است با ۴
ضربدرِ فاکتوریلِ ۳، و...» تا به فاکتوریلِ ۱ برسد که پاسخش روشن است: ۱.

\begin{warn}
هر تابعِ بازگشتی به یک \textbf{شرطِ توقف} (پایهٔ بازگشت) نیاز دارد جایی که
دیگر خودش را صدا نمی‌زند. در فاکتوریل، این شرط \code{ن <= 1} است. اگر این شرط
را فراموش کنید، تابع تا بی‌نهایت خودش را صدا می‌زند و برنامه از کار می‌افتد. این
خطا مانندِ همان «حلقهٔ بی‌پایانِ» فصلِ پیش است.
\end{warn}

\begin{deep}[نگاهی به درون: بازگشت چگونه کار می‌کند]
هر بار که تابعی صدا زده می‌شود چه خودش، چه دیگری رایانه یک «قاب» تازه
روی \emph{پشتهٔ فراخوانی} (call stack) می‌سازد که پارامترها و متغیرهای آن
فراخوانی را نگه می‌دارد. در فاکتوریلِ ۵، پنج قاب روی هم چیده می‌شوند تا به پایهٔ
بازگشت برسیم؛ سپس یکی‌یکی باز می‌گردند و ضرب‌ها انجام می‌شود. به همین دلیل
بازگشتِ بیش‌از‌اندازه عمیق می‌تواند پشته را پُر کند نکته‌ای که در فصلِ ۲۰
به آن بازمی‌گردیم.
\end{deep}

\section{ترتیبِ تعریفِ توابع}

در نمونه‌های این کتاب، توابعِ کمکی را معمولاً \emph{پیش} از تابعِ \code{اصلی}
می‌نویسیم. اما این تنها یک قراردادِ زیباست، نه یک اجبار. سلام برنامه را در دو
گذر بررسی می‌کند: نخست همهٔ توابع را \emph{می‌شناسد}، سپس بدنه‌ها را وارسی
می‌کند. به همین دلیل می‌توانید تابعی را \emph{پیش} از تعریفش هم صدا بزنید ---
چیزی که گاهی «اعلانِ پیش‌رو» نامیده می‌شود و در زبان‌های دیگر دردسرساز است، اما
در سلام به‌طورِ طبیعی کار می‌کند.

\section{ترکیبِ توابع}

زیباییِ توابع آن‌جاست که خروجیِ یکی را می‌توان \emph{ورودیِ} دیگری کرد. در
مثالِ نخستِ این فصل دیدیم:

\begin{salam}
چاپ("بیشینه:", بیشینه(توان(2, 4), توان(3, 2)))
\end{salam}

اینجا نخست \code{توان(2, 4)} برابرِ ۱۶ و \code{توان(3, 2)} برابرِ ۹ محاسبه
می‌شوند، سپس \code{بیشینه(16, 9)} برابرِ ۱۶ می‌شود. این «لانه‌سازیِ» توابع، پایهٔ
ساختنِ برنامه‌های بزرگ از قطعاتِ کوچک است.

\section{خلاصهٔ فصل}

\begin{itemize}
  \item تابع، بسته‌ای نام‌دار از دستورهاست که از تکرار جلوگیری می‌کند و کد را
        خوانا می‌سازد.
  \item پارامترها ورودیِ تابع‌اند و هنگامِ صدا زدن مقدار می‌گیرند.
  \item \code{بازگشت} خروجی را می‌فرستد و از تابع خارج می‌شود.
  \item پارامترهای پایانی می‌توانند \emph{مقدارِ پیش‌فرض} داشته باشند
        (\code{ب صحیح۳۲ = 0}) که فراخواننده می‌تواند آن را ننویسد.
  \item توابع می‌توانند بدونِ خروجی باشند.
  \item تابعِ بازگشتی خودش را صدا می‌زند و به شرطِ توقف نیاز دارد.
  \item خروجیِ یک تابع می‌تواند ورودیِ تابعِ دیگر باشد.
\end{itemize}

\section{تمرین‌ها}

\exercise تابعی به نامِ \code{مربع} بنویسید که یک عدد بگیرد و مربعِ آن را
برگرداند.

\exercise تابعی به نامِ \code{کمینه} بنویسید که دو عدد بگیرد و کوچک‌ترشان را
برگرداند.

\exercise تابعِ بازگشتیِ \code{فیبوناچی} را بنویسید
(\code{فیبوناچی(ن) = فیبوناچی(ن-۱) + فیبوناچی(ن-۲)}) با شرطِ توقفِ مناسب.

\exercise تابعِ \code{توان(پایه صحیح۳۲, نما صحیح۳۲ = 2) صحیح۳۲} را بنویسید که
\code{پایه} را به توانِ \code{نما} برساند. آن را به‌صورتِ \code{توان(5)} (باید ۲۵
شود) و \code{توان(2, 10)} (باید ۱۰۲۴ شود) صدا بزنید. سپس پارامتری \emph{بدونِ}
پیش‌فرض پس از \code{نما} اضافه کنید و خطای کامپایلر را بخوانید.

\exercise تابعی بنویسید که یک عدد بگیرد و بگوید اول است یا نه (راهنمایی: با یک
حلقه بررسی کنید که آیا مقسوم‌علیهی غیر از ۱ و خودش دارد).

\exercise تابعی به نامِ \code{ب.م.م} بنویسید که بزرگ‌ترین مقسوم‌علیهِ مشترکِ دو
عدد را با روشِ اقلیدس حساب کند.
