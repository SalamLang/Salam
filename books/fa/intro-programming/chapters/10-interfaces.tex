\chapter{واسط‌ها و چندریختی}%
\label{ch:interfaces}

در فصلِ ۹ دیدیم که هر ساختار می‌تواند متدهای خود را داشته باشد. اما گاهی چند
ساختارِ متفاوت، \emph{رفتارِ} مشترکی دارند: هم مربع و هم دایره «مساحت» دارند، هر
چند فرمولشان فرق می‌کند. چگونه می‌توان با همهٔ آن‌ها به یک شیوهٔ یکسان کار کرد؟
پاسخ، \textbf{واسط} است.

\section{مسئله: رفتارِ مشترک، پیاده‌سازیِ متفاوت}

فرض کنید چند شکلِ هندسی دارید و می‌خواهید تابعی بنویسید که مساحتِ \emph{هر}
شکلی را چاپ کند بی‌آنکه برای هر شکل، تابعی جدا بنویسید. مشکل اینجاست که مربع
و مستطیل دو نوعِ \emph{متفاوت}اند؛ یک تابع چگونه می‌تواند هر دو را بپذیرد؟

\textbf{واسط} (interface) راهِ حل است: قراردادی که می‌گوید «هر نوعی که این
متدها را داشته باشد، یک شکل به‌شمار می‌آید.»

\section{تعریف و پیاده‌سازیِ واسط}

به این مثالِ واقعی نگاه کنید که قراردادِ «شکل» را تعریف و سپس دو نوع را با آن
هماهنگ می‌کند:

\begin{salam}
واسط شکل:
    تابع مساحت() : صحیح
    تابع نام() : رشته
تمام
ساختار مربع:
    ضلع : صحیح = 0
    تابع مساحت() : صحیح: بازگشت این.ضلع * این.ضلع تمام
    تابع نام() : رشته: بازگشت "مربع" تمام
تمام
ساختار مستطیل:
    طول : صحیح = 0
    عرض : صحیح = 0
    تابع مساحت() : صحیح: بازگشت این.طول * این.عرض تمام
    تابع نام() : رشته: بازگشت "مستطیل" تمام
تمام

تابع توصیف(ش : dyn شکل):
    چاپ(ش.نام(), "->", ش.مساحت())
تمام

تابع اصلی:
    توصیف(مربع { ضلع = 5 })
    توصیف(مستطیل { طول = 4, عرض = 6 })
تمام
\end{salam}

این مثال چند بخشِ کلیدی دارد:

\subsection{اعلانِ واسط}

\begin{salam}
واسط شکل:
    تابع مساحت() : صحیح
    تابع نام() : رشته
تمام
\end{salam}

\code{واسط شکل} می‌گوید: «هر چیزی که بخواهد یک \emph{شکل} باشد، باید دو متد
داشته باشد: \code{مساحت} که یک عدد می‌دهد، و \code{نام} که یک رشته می‌دهد.»
دقت کنید که در واسط فقط \emph{امضا}ی متدها آمده، نه بدنه‌شان. واسط می‌گوید
\emph{چه} رفتاری لازم است، نه \emph{چگونه}.

\subsection{پیاده‌سازیِ واسط}

\code{مربع} و \code{مستطیل} هر دو متدهای \code{مساحت} و \code{نام} را دارند، پس
هر دو واسطِ \code{شکل} را برآورده می‌کنند. در سلام لازم نیست صریحاً بنویسید
«مربع از شکل پیروی می‌کند»؛ همین که متدهای لازم را داشته باشد، کافی است.

\subsection{استفادهٔ همگانی با dyn}

\begin{salam}
تابع توصیف(ش : dyn شکل):
    چاپ(ش.نام(), "->", ش.مساحت())
تمام
\end{salam}

اینجا جادو رخ می‌دهد. تابعِ \code{توصیف} یک پارامتر از نوعِ \code{dyn شکل}
می‌گیرد یعنی «هر چیزی که یک شکل باشد». حالا می‌توانیم هم مربع و هم مستطیل را
به آن بدهیم و هر کدام متدِ خودش را اجرا می‌کند. خروجی:

\begin{salamen}[language={}]
مربع -> 25
مستطیل -> 24
\end{salamen}

به این توانایی اینکه یک تابع با انواعِ گوناگون کار کند و هر نوع رفتارِ خود
را نشان دهد \textbf{چندریختی} (polymorphism) می‌گویند؛ واژه‌ای یونانی به
معنای «چند شکلی».

\begin{deep}[نگاهی به درون: dyn چگونه کار می‌کند]
وقتی یک مقدار را به‌صورتِ \code{dyn شکل} می‌فرستید، سلام در پسِ پرده یک
«اشاره‌گرِ چاق» (fat pointer) می‌سازد: یک اشاره به خودِ داده، و یک اشاره به
جدولی از متدهای آن نوع (که به آن \emph{جدولِ مجازی} یا vtable می‌گویند). وقتی
\code{ش.مساحت()} را صدا می‌زنید، سلام در زمانِ اجرا از این جدول، متدِ درستِ
همان نوع را پیدا و اجرا می‌کند. به همین دلیل به \code{dyn} «ارسالِ پویا»
(dynamic dispatch) می‌گویند: تصمیمِ اینکه کدام متد اجرا شود، در \emph{زمانِ
اجرا} گرفته می‌شود، نه زمانِ کامپایل.
\end{deep}

\section{چرا واسط‌ها ارزشمندند؟}

تصور کنید فردا بخواهید شکلِ تازه‌ای مثلاً «مثلث» بیفزایید. کافی است یک
ساختارِ \code{مثلث} با متدهای \code{مساحت} و \code{نام} بسازید. تابعِ
\code{توصیف} \emph{بدونِ هیچ تغییری} با مثلث هم کار می‌کند! این یعنی کدِ شما
\textbf{گسترش‌پذیر} است: می‌توانید انواعِ تازه بیفزایید بی‌آنکه کدِ موجود را
دست بزنید.

\begin{tip}
واسط‌ها به شما اجازه می‌دهند کد را بر پایهٔ \emph{رفتار} بنویسید، نه بر پایهٔ
نوعِ مشخص. این یکی از مهم‌ترین ابزارهای طراحیِ نرم‌افزارِ بزرگ و قابلِ نگه‌داری
است. هر جا دیدید که یک تابع را برای چند نوعِ مشابه تکرار می‌کنید، احتمالاً یک
واسط می‌تواند آن تکرار را از میان بردارد.
\end{tip}

\section{واسط در برابرِ کلی‌سازی}

سلام دو راه برای نوشتنِ کدی که با انواعِ گوناگون کار کند دارد: \emph{واسط}
(این فصل) و \emph{کلی‌سازی} (فصلِ بعد). تفاوتشان ظریف اما مهم است:

\begin{itemize}
  \item \textbf{واسط با \code{dyn}:} انعطاف‌پذیر است؛ می‌توانید انواعِ مختلف
        را در یک بردار بریزید و در زمانِ اجرا تصمیم بگیرید. اما اندکی هزینهٔ
        اجرایی دارد (به‌خاطرِ همان جدولِ مجازی).
  \item \textbf{کلی‌سازی:} در زمانِ کامپایل به کدِ مخصوصِ هر نوع تبدیل می‌شود،
        پس هیچ هزینهٔ اجرایی ندارد و سریع‌تر است؛ اما کمی انعطافِ کمتری دارد.
\end{itemize}

در فصلِ بعد کلی‌سازی را خواهیم دید و آن‌گاه می‌توانید آگاهانه میانِ این دو
انتخاب کنید.

\section{خلاصهٔ فصل}

\begin{itemize}
  \item واسط قراردادی است که می‌گوید یک نوع چه متدهایی باید داشته باشد.
  \item هر ساختاری که متدهای لازم را داشته باشد، واسط را برآورده می‌کند ---
        بی‌نیاز به اعلانِ صریح.
  \item با \code{dyn واسط} می‌توان تابعی نوشت که با هر نوعِ سازگار کار کند.
  \item چندریختی یعنی یک تابع با انواعِ گوناگون کار کند و هر نوع رفتارِ خود را
        نشان دهد.
  \item واسط‌ها کد را گسترش‌پذیر می‌کنند.
\end{itemize}

\section{تمرین‌ها}

\exercise واسطِ \code{شکل} را گسترش دهید و یک ساختارِ \code{دایره} بیفزایید که
آن را برآورده کند، سپس آن را به \code{توصیف} بدهید.

\exercise واسطی به نامِ \code{جاندار} با متدِ \code{صدا() رشته} بسازید و
ساختارهای \code{گربه} و \code{سگ} را برایش پیاده کنید.

\exercise تابعی بنویسید که یک \code{dyn شکل} بگیرد و بگوید مساحتش از ۱۰۰ بیشتر
است یا نه.

\exercise واسطی به نامِ \code{قابل‌چاپ} با متدِ \code{به‌رشته() رشته} بسازید و
دو نوعِ متفاوت را برایش پیاده کنید.
