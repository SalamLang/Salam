\chapter{کلی‌سازی}
\label{ch:generics}

تصور کنید تابعی نوشته‌اید که بزرگ‌ترِ دو عددِ صحیح را برمی‌گرداند. حالا همان را
برای اعدادِ اعشاری هم می‌خواهید. آیا باید تابع را دوباره با همان منطق و فقط
نوعِ متفاوت بنویسید؟ خوشبختانه نه. \textbf{کلی‌سازی} (generics) به ما اجازه
می‌دهد یک‌بار کدی بنویسیم که با \emph{هر} نوعی کار کند.

\section{مسئله: تکرارِ یک منطق برای انواعِ گوناگون}

بدونِ کلی‌سازی، ناچار بودیم چنین بنویسیم:

\begin{salam}
تابع بیشینه صحیح(الف : صحیح, ب : صحیح) : صحیح:
    اگر الف > ب: بازگشت الف تمام
    بازگشت ب
تمام

تابع بیشینه اعشاری(الف : اعشار۶۴, ب : اعشار۶۴) : اعشار۶۴:
    اگر الف > ب: بازگشت الف تمام
    بازگشت ب
تمام
\end{salam}

این دو تابع \emph{منطقِ یکسانی} دارند و تنها در نوع فرق می‌کنند. این تکرار هم
خسته‌کننده است و هم خطاخیز: اگر روزی منطق را اصلاح کنید، باید در هر دو جا تغییر
دهید.

\section{راهِ حل: پارامترِ نوع}

با کلی‌سازی، نوع را هم مانندِ یک پارامتر می‌گیریم. این مثالِ واقعی را ببینید:

\begin{salam}
تابع بیشینه<ت>(الف : ت, ب : ت) : ت:
    اگر الف > ب:
        بازگشت الف
    تمام
    بازگشت ب
تمام
ساختار جعبه<ت>:
    مقدار : ت
تمام

تابع اصلی:
    چاپ(بیشینه(3, 9))
    چاپ(بیشینه(2.5, 1.5))
    متغیر ج : جعبه<صحیح> = جعبه { مقدار = 42 }
    چاپ(ج.مقدار)
تمام
\end{salam}

به \code{بیشینه<ت>} دقت کنید. آن \code{ت} درونِ \code{<\,>} یک \textbf{پارامترِ
نوع} است یک نوعِ «جای‌خالی» که هنوز مشخص نیست. سپس در امضای تابع،
پارامترها و خروجی همگی از همان نوعِ \code{ت} هستند. معنایش این است: «هر دو
ورودی و خروجی، از یک نوعِ یکسان‌اند هر نوعی که باشد.»

حالا وقتی می‌نویسیم:
\begin{itemize}
  \item \code{بیشینه(3, 9)}: سلام می‌بیند که ورودی‌ها صحیح‌اند، پس \code{ت} را
        \code{صحیح} می‌گیرد.
  \item \code{بیشینه(2.5, 1.5)}: ورودی‌ها اعشاری‌اند، پس \code{ت} را
        \code{اعشاری} می‌گیرد.
\end{itemize}

یک تابع، دو کاربرد بی‌هیچ تکراری!

\begin{note}
دقت کنید که هنگامِ \emph{صدا زدنِ} \code{بیشینه}، لازم نبود نوع را بنویسیم؛ سلام
آن را از روی ورودی‌ها \emph{حدس} زد. این را «استنتاجِ نوع» می‌گویند و کد را
تمیزتر می‌کند.
\end{note}

\section{ساختارهای کلی}

نه‌تنها توابع، بلکه \emph{ساختارها} هم می‌توانند کلی باشند. در مثالِ بالا:

\begin{salam}
ساختار جعبه<ت>:
    مقدار : ت
تمام
\end{salam}

\code{جعبه<ت>} یک ساختارِ کلی است: جعبه‌ای که می‌تواند \emph{هر} نوعی را در خود
نگه دارد. وقتی می‌نویسیم \code{جعبه<صحیح>}، یک جعبهٔ مخصوصِ اعدادِ صحیح
می‌سازیم. می‌توانستیم \code{جعبه<رشته>} هم بسازیم تا متن نگه دارد.

\begin{tip}
بردار (\code{وکتور<T>}) و نگاشت (\code{نگاشت<K, V>}) که در فصل‌های پیش دیدیم،
خودشان ساختارهای کلی‌اند! به همین دلیل می‌توانستیم هم \code{وکتور<صحیح>} و هم
\code{وکتور<رشته>} بسازیم. اکنون می‌دانید که در پسِ این انعطاف، همین کلی‌سازی
نهفته است.
\end{tip}

\section{کلی‌سازی با چند پارامترِ نوع}

یک ساختار یا تابع می‌تواند بیش از یک پارامترِ نوع داشته باشد. برای نمونه، یک
«جفت» که دو مقدارِ احتمالاً متفاوت را نگه می‌دارد:

\begin{salam}
ساختار جفت<آ, ب>:
    اول : آ
    دوم : ب
تمام

تابع اصلی:
    متغیر ج : جفت<رشته, صحیح> = جفت { اول = "سن", دوم = 30 }
    چاپ(ج.اول, ج.دوم)
تمام
\end{salam}

اینجا \code{آ} و \code{ب} دو پارامترِ نوعِ مستقل‌اند؛ پس می‌توان جفتی ساخت که
عضوِ نخستش رشته و عضوِ دومش عدد باشد. همان‌گونه که \code{نگاشت<K, V>} از کلید و
مقدارِ متفاوت بهره می‌برد.

\section{چرا کلی‌سازی سریع است؟}

\begin{deep}[نگاهی به درون: تک‌ریخت‌سازی]
سلام کلی‌سازی را با روشی به نامِ \emph{تک‌ریخت‌سازی} (monomorphization) پیاده
می‌کند. وقتی شما \code{بیشینه(3, 9)} و \code{بیشینه(2.5, 1.5)} را می‌نویسید،
سلام در زمانِ کامپایل \emph{دو نسخهٔ جداگانه} از تابع می‌سازد: یکی مخصوصِ صحیح و
یکی مخصوصِ اعشاری. یعنی همان کدی که در آغازِ این فصل دستی نوشتیم، این بار به‌طورِ
خودکار ساخته می‌شود.

نتیجه این است که کلی‌سازی در سلام \emph{هیچ هزینهٔ اجرایی} ندارد: کدِ نهایی
دقیقاً به سرعتِ کدی است که برای هر نوع جدا نوشته باشید. شما راحتیِ نوشتنِ یک
تابع را دارید و رایانه سرعتِ چند تابعِ تخصصی را. این یکی از زیباترین معامله‌های
برنده-برنده در طراحیِ زبان است.
\end{deep}

\section{کلی‌سازی در برابرِ واسط: کدام را برگزینیم؟}

اکنون که هر دو ابزار را دیدید، این راهنمای ساده کمکتان می‌کند:

\begin{center}
\begin{tabular}{l l}
\toprule
\textbf{کلی‌سازی را برگزینید اگر...} & \textbf{واسطِ \code{dyn} را برگزینید اگر...} \\
\midrule
سرعت بسیار مهم است & به انعطافِ زمانِ اجرا نیاز دارید \\
نوع‌ها در زمانِ کامپایل معلوم‌اند & می‌خواهید انواعِ مختلف را در یک بردار بریزید \\
منطقِ یکسان برای انواعِ گوناگون & رفتارِ مشترک با پیاده‌سازیِ متفاوت \\
\bottomrule
\end{tabular}
\end{center}

\begin{tip}
قاعدهٔ سرانگشتی: اگر شک دارید، کلی‌سازی را برگزینید. در بیشترِ موارد سریع‌تر و
ایمن‌تر است. تنها وقتی به \code{dyn} روی بیاورید که واقعاً به ذخیرهٔ انواعِ
متفاوت در کنارِ هم یا تصمیم‌گیریِ زمانِ اجرا نیاز داشته باشید.
\end{tip}

\section{خلاصهٔ فصل}

\begin{itemize}
  \item کلی‌سازی به ما اجازه می‌دهد یک‌بار کدی بنویسیم که با هر نوعی کار کند.
  \item پارامترِ نوع (مثلِ \code{ت}) درونِ \code{<\,>} نوشته می‌شود.
  \item هم توابع و هم ساختارها می‌توانند کلی باشند.
  \item می‌توان چند پارامترِ نوع داشت: \code{جفت<آ, ب>}.
  \item سلام نوع را از روی ورودی‌ها استنتاج می‌کند.
  \item کلی‌سازی با تک‌ریخت‌سازی پیاده می‌شود و هیچ هزینهٔ اجرایی ندارد.
\end{itemize}

\section{تمرین‌ها}

\exercise تابعِ کلیِ \code{کمینه<ت>} را بنویسید که کوچک‌ترِ دو مقدار را
برگرداند.

\exercise تابعِ کلیِ \code{جابجا<ت>} را بنویسید که... (راهنمایی: به مفهومِ
ارجاع در فصلِ ۱۳ نیاز دارید؛ فعلاً دو مقدار بگیرید و آن‌ها را در یک \code{جفت}ِ
جابه‌جا‌شده برگردانید).

\exercise ساختارِ کلیِ \code{پشته<ت>} را با متدهای \code{بیفزا} و
\code{دربیاور} بسازید (می‌توانید درونش از یک بردار استفاده کنید).

\exercise ساختارِ \code{جفت<آ, ب>} را گسترش دهید و متدی بیفزایید که جای دو عضو
را عوض کند و یک \code{جفت<ب, آ>} برگرداند.

\exercise توضیح دهید چرا برای ساختنِ یک «فهرستِ نام‌ها و سن‌ها»، \code{نگاشت}
بهتر از دو بردارِ جداگانه است.
