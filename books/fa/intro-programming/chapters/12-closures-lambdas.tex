% ===================== فصل ۱۲ =====================
\chapter{بستارها و عبارت‌های لامبدا}
\label{ch:lambdas}

تا اینجا توابع را همیشه با نام می‌شناختیم. اما در سلام، تابع خودش یک \emph{مقدار}
است: می‌توان آن را در متغیر گذاشت، به تابعِ دیگر داد، یا از یک تابع
بازگرداند. در این فصل با \textbf{توابعِ بی‌نام} (لامبدا) و \textbf{بستارها}
آشنا می‌شویم --- ابزارهایی که برنامه‌نویسی را انعطاف‌پذیرتر می‌کنند.

\section{تابع به‌مثابهِ مقدار}

به این مثالِ واقعی نگاه کنید. اینجا توابع را در متغیر می‌گذاریم و حتی به تابعِ
دیگری می‌دهیم:

\begin{salam}
تابع اعمال(ف func(صحیح۳۲) صحیح۳۲, مقدار صحیح۳۲) صحیح۳۲:
    بازگشت ف(مقدار)
پایان
تابع اصلی:
    دوبرابر func(صحیح۳۲) صحیح۳۲ = (ن صحیح۳۲) => ن * 2
    مربع func(صحیح۳۲) صحیح۳۲ = (ن صحیح۳۲) => ن * ن
    چاپ(اعمال(دوبرابر, 21))
    چاپ(اعمال(مربع, 9))
    چاپ(دوبرابر(مربع(3)))
پایان
\end{salam}

چند چیزِ تازه اینجا هست:

\subsection{نوعِ تابع}

\begin{salam}
func(صحیح۳۲) صحیح۳۲
\end{salam}

این یک \textbf{نوع} است --- مانندِ \code{صحیح} یا \code{رشته}، اما نوعِ یک
\emph{تابع}. خوانده می‌شود: «تابعی که یک عددِ صحیح می‌گیرد و یک عددِ صحیح
برمی‌گرداند.» پس \code{دوبرابر} متغیری است که نه عدد، بلکه یک \emph{تابع} در
آن نگه‌داشته می‌شود.

\subsection{عبارتِ لامبدا}

\begin{salam}
(ن صحیح۳۲) => ن * 2
\end{salam}

این یک \textbf{لامبدا} است: یک تابعِ کوچکِ بی‌نام. ساختارش ساده است: درونِ
پرانتز پارامترها، سپس نشانهٔ \code{=>}، و سپس عبارتی که نتیجه را می‌دهد. این
لامبدا یک عدد می‌گیرد و دوبرابرش را برمی‌گرداند --- درست مانندِ یک تابعِ معمولی،
اما بدونِ نام و در یک خط.

\subsection{تابعی که تابع می‌گیرد}

تابعِ \code{اعمال} در پارامترِ نخستش یک \emph{تابع} می‌گیرد (\code{ف}) و آن را
روی \code{مقدار} اجرا می‌کند. به همین دلیل توانستیم هم \code{دوبرابر} و هم
\code{مربع} را به آن بدهیم. توابعی که تابعِ دیگر می‌گیرند یا برمی‌گردانند را
\textbf{توابعِ مرتبهٔ بالاتر} می‌نامند و در پردازشِ داده بسیار کاربرد دارند.

\begin{note}
به آخرین خط دقت کنید: \code{دوبرابر(مربع(3))}. نخست \code{مربع(3)} برابرِ ۹، سپس
\code{دوبرابر(9)} برابرِ ۱۸ می‌شود. چون این‌ها متغیرهایی حاوی تابع‌اند، می‌توان
مستقیم صدایشان زد --- درست مانندِ توابعِ معمولی.
\end{note}

\section{بستار: لامبدایی که محیطش را به خاطر می‌سپارد}

اینجا به یکی از قدرتمندترین ایده‌های برنامه‌نویسی می‌رسیم. یک لامبدا می‌تواند
متغیرهای \emph{بیرونِ} خودش را «به یاد بسپارد». به چنین لامبدایی \textbf{بستار}
(closure) می‌گویند. این مثالِ واقعی را ببینید:

\begin{salam}
تابع سازنده_جمع(ن i32) func(i32) i32:
    بازگشت (x i32) => x + ن
پایان
تابع اصلی:
    دوبرابر func(i32) i32 = (n i32) => n * 2
    چاپ دوبرابر(21)
    جمع۱۰ خودکار = سازنده_جمع(10)
    چاپ جمع۱۰(5)
پایان
\end{salam}

تابعِ \code{سازنده\_جمع} یک عدد (\code{ن}) می‌گیرد و یک \emph{لامبدا}
برمی‌گرداند. اما دقت کنید: آن لامبدا --- \code{(x i32) => x + ن} --- از
\code{ن} استفاده می‌کند که پارامترِ تابعِ بیرونی است! وقتی
\code{سازنده\_جمع(10)} را صدا می‌زنیم، لامبدای بازگشتی مقدارِ \code{ن = 10} را
\emph{با خود نگه می‌دارد}. پس \code{جمع۱۰(5)} برابرِ $5 + 10 = 15$ می‌شود.

این یعنی \code{جمع۱۰} یک تابعِ تازه است که «۱۰ را به یاد دارد». اگر
\code{سازنده\_جمع(100)} می‌ساختیم، تابعی می‌داشتیم که ۱۰۰ را به یاد می‌آورد.

\section{بستار با چند دستور}

لامبداهایی که دیدیم یک عبارت داشتند. اما گاهی به چند خط منطق نیاز داریم. برای
این، شکلِ بلندترِ لامبدا را داریم که با \code{=> :} باز و با \code{پایان} بسته
می‌شود. این مثالِ واقعی یک \emph{شمارنده} می‌سازد:

\begin{salam}
تابع ساخت_شمارنده() func() صحیح۳۲:
    متغیر ن صحیح۳۲ = 0
    بازگشت () => :
        ن = ن + 1
        بازگشت ن
    پایان
پایان
تابع اصلی:
    بعدی func() صحیح۳۲ = ساخت_شمارنده()
    چاپ بعدی(), بعدی(), بعدی(), بعدی()
پایان
\end{salam}

اینجا اتفاقِ جالبی می‌افتد. \code{ساخت\_شمارنده} متغیرِ \code{ن} را می‌سازد و
یک بستار برمی‌گرداند که آن را یکی زیاد و سپس بازمی‌گرداند. هر بار که
\code{بعدی()} را صدا می‌زنیم، همان \code{ن} --- که \emph{میانِ فراخوانی‌ها زنده
می‌ماند} --- یکی بیشتر می‌شود. پس خروجی \code{1 2 3 4} است.

\begin{deep}[نگاهی به درون: بستار حافظه را کجا نگه می‌دارد؟]
وقتی یک بستار متغیری از محیطِ بیرون را می‌گیرد، سلام آن متغیر را در \emph{پشته}
نمی‌گذارد (که با پایانِ تابع از میان می‌رود)، بلکه در \emph{پشتهٔ آزاد} (heap)
نگه می‌دارد و یک «محیطِ بسته‌شده» می‌سازد که همراهِ خودِ تابع زندگی می‌کند. به
همین دلیل است که \code{ن} حتی پس از پایان‌یافتنِ \code{ساخت\_شمارنده} همچنان
زنده می‌ماند. این مکانیزم --- گرفتنِ متغیرها و نگه‌داشتنشان روی پشتهٔ آزاد ---
قلبِ کارِ بستارهاست.
\end{deep}

\begin{tip}
بستارها به‌ویژه برای ساختنِ «کارخانه‌های تابع» (مانندِ \code{سازنده\_جمع}) و
نگه‌داشتنِ حالتِ خصوصی (مانندِ شمارنده) عالی‌اند. در فصلِ کتابخانهٔ استاندارد
خواهیم دید که بسیاری از توابعِ پردازشِ داده، لامبدا می‌گیرند تا رفتارشان را
سفارشی کنید.
\end{tip}

\section{خلاصهٔ فصل}

\begin{itemize}
  \item در سلام، تابع یک مقدار است: می‌توان آن را در متغیر گذاشت، پاس داد و
        بازگرداند.
  \item نوعِ تابع چنین نوشته می‌شود: \code{func(ورودی) خروجی}.
  \item لامبدا تابعی بی‌نام است: \code{(پارامتر) => عبارت}.
  \item برای چند دستور، شکلِ بلند را به‌کار می‌بریم: \code{() => :} \ldots
        \code{پایان}.
  \item بستار لامبدایی است که متغیرهای محیطِ خود را به یاد می‌سپارد.
  \item توابعِ مرتبهٔ بالاتر، تابعِ دیگر را می‌گیرند یا برمی‌گردانند.
\end{itemize}

\section{تمرین‌ها}

\exercise یک لامبدا بسازید که عددی بگیرد و مکعبش را برگرداند، و آن را در یک
متغیر بگذارید و صدا بزنید.

\exercise تابعی به نامِ \code{سازنده\_ضرب} بنویسید که عددی بگیرد و یک بستار
برگرداند که هر ورودی را در آن عدد ضرب کند.

\exercise با کمکِ \code{ساخت\_شمارنده}، دو شمارندهٔ مستقل بسازید و نشان دهید که
هر کدام حالتِ خودش را دارد.

\exercise تابعِ \code{اعمال\_دوبار} را بنویسید که یک تابع و یک مقدار بگیرد و
تابع را \emph{دو بار} پشتِ سرِ هم روی مقدار اجرا کند.

\exercise بستاری بسازید که مجموعِ همهٔ اعدادی را که تا کنون به آن داده‌اید نگه
دارد و در هر فراخوانی، مجموعِ تازه را برگرداند.
