\chapter{مدیریتِ خطا و اجرای به‌تعویق‌افتاده}
\label{ch:defer}

برنامه‌های واقعی همیشه در مسیرِ خوش پیش نمی‌روند: فایل ممکن است یافت نشود،
ورودیِ کاربر ممکن است نادرست باشد، یا منبعی که گرفته‌ایم باید آزاد شود. در این
فصل می‌آموزیم چگونه برنامه‌ای بنویسیم که با این موقعیت‌ها \emph{تمیز} و
\emph{مطمئن} برخورد کند.

\section{تعلیق: کاری که حتماً باید انجام شود}

گاهی کاری هست که باید \emph{در پایانِ} یک تابع انجام شود هرچه پیش بیاید.
مثلاً اگر فایلی را باز کرده‌اید، باید آن را ببندید؛ اگر حافظه‌ای گرفته‌اید، باید
آزادش کنید. واژهٔ \code{تعلیق} (معادلِ \code{defer}) دقیقاً برای همین است. به
این مثالِ واقعیِ کوتاه نگاه کنید:

\begin{salam}
تابع اصلی:
    تعلیق چاپ "پاکسازی"
    چاپ "شروع"
    چاپ "پایان"
پایان
\end{salam}

شاید انتظار داشته باشید خروجی به ترتیبِ نوشتن باشد، اما چنین نیست:

\begin{salamen}[language={}]
شروع
پایان
پاکسازی
\end{salamen}

دستوری که با \code{تعلیق} نشانه‌گذاری می‌شود، \emph{بی‌درنگ اجرا نمی‌شود}؛
بلکه کنار گذاشته می‌شود تا \emph{درست پیش از پایانِ تابع} اجرا گردد. پس هرچند
«پاکسازی» را در خطِ نخست نوشتیم، در پایان اجرا شد.

\section{چرا تعلیق این‌قدر مفید است؟}

ارزشِ واقعیِ \code{تعلیق} وقتی روشن می‌شود که تابع چند راهِ خروج دارد. این مثالِ
واقعی را ببینید:

\begin{salam}
تابع کار():
    تعلیق چاپ("۳. منبع آزاد شد")
    چاپ("۱. شروع کار")
    چاپ("۲. انجام کار")
پایان

تابع اصلی:
    کار()
    چاپ("۴. پایان برنامه")
پایان
\end{salam}

خروجی به ترتیبِ ۱، ۲، ۳، ۴ است. نکته اینجاست: \emph{هر} مسیری که تابع از آن
خارج شود چه به‌طورِ طبیعی، چه با \code{بازگشت} زودهنگام دستورِ
\code{تعلیق} اجرا می‌شود. به همین دلیل، \code{تعلیق} بهترین جا برای «آزادسازیِ
منابع» است: کافی است \emph{بلافاصله} پس از گرفتنِ منبع، آزادسازی‌اش را
\code{تعلیق} کنید و دیگر هرگز نگرانِ فراموش‌کردنش نباشید.

\begin{tip}
الگوی طلایی: هر بار منبعی گرفتید (فایل باز کردید، حافظه گرفتید، قفلی گرفتید)،
بی‌درنگ در خطِ بعد آزادسازی‌اش را \code{تعلیق} کنید. این‌گونه گرفتن و آزادکردن
کنارِ هم می‌مانند و خواندنِ کد آسان می‌شود.
\begin{salam}
// شبه‌کد
ف = باز_کن("داده.txt")
تعلیق ف.ببند()
// ... هر کاری با ف، حتی با چند بازگشتِ زودهنگام ...
\end{salam}
\end{tip}

\section{چند تعلیق و ترتیبِ اجرا}

اگر چند \code{تعلیق} داشته باشید، به ترتیبِ \emph{معکوس} اجرا می‌شوند یعنی
آخرین تعلیق نخست اجرا می‌شود:

\begin{salam}
تابع اصلی:
    تعلیق چاپ "یک"
    تعلیق چاپ "دو"
    تعلیق چاپ "سه"
    چاپ "بدنه"
پایان
\end{salam}

خروجی:

\begin{salamen}[language={}]
بدنه
سه
دو
یک
\end{salamen}

\begin{deep}[نگاهی به درون: چرا ترتیب معکوس است؟]
\code{تعلیق}ها روی یک \emph{پشته} انباشته می‌شوند. همان‌طور که در فصلِ بردارها
دیدیم، در پشته «آخرین چیزی که گذاشتی، نخستین چیزی است که برمی‌داری». این ترتیبِ
معکوس در عمل دقیقاً همان چیزی است که می‌خواهیم: اگر منبعِ «الف» را پیش از «ب»
گرفته‌اید، طبیعی است که «ب» را پیش از «الف» آزاد کنید درست مثلِ بازکردنِ
لایه‌های یک پیاز از بیرون به درون.
\end{deep}

\section{مدیریتِ خطا با مقدارِ بازگشتی}

سلام رویکردی \emph{صریح} به خطاها دارد: به‌جای آنکه خطاها به‌طورِ پنهان رخ
دهند، تابع‌ها معمولاً یک \emph{نشانه} برمی‌گردانند که می‌گوید کار موفق بود یا
نه. ساده‌ترین شکل، بازگرداندنِ یک مقدارِ منطقی یا یک کدِ وضعیت است:

\begin{salam}
تابع تقسیمِ_امن(الف صحیح۳۲, ب صحیح۳۲, نتیجه &صحیح۳۲) منطقی:
    اگر ب == 0:
        بازگشت نادرست
    پایان
    نتیجه = الف / ب
    بازگشت درست
پایان

تابع اصلی:
    متغیر خروجی صحیح۳۲ = 0
    اگر تقسیمِ_امن(10, 2, خروجی):
        چاپ "نتیجه:", خروجی
    وگرنه:
        چاپ "خطا: تقسیم بر صفر"
    پایان
    اگر تقسیمِ_امن(5, 0, خروجی):
        چاپ "نتیجه:", خروجی
    وگرنه:
        چاپ "خطا: تقسیم بر صفر"
    پایان
پایان
\end{salam}

اینجا تابع \code{درست} یا \code{نادرست} برمی‌گرداند تا بگوید موفق شد یا نه، و
نتیجهٔ واقعی را از راهِ پارامترِ \emph{ارجاعی} \code{نتیجه \&صحیح۳۲} بیرون
می‌دهد. (نشانهٔ \code{\&} یک \emph{ارجاع} است که به تابع اجازه می‌دهد متغیرِ
بیرون را تغییر دهد؛ آن را در فصلِ ۲۰ کامل‌تر می‌بینیم.) فراخواننده
\emph{موظف} است وضعیت را بررسی کند و همین صراحت، خطاها را آشکار و کنترل‌پذیر
می‌کند.

\begin{warn}
در زبان‌هایی که خطاها را پنهان مدیریت می‌کنند، آسان است که فراموش کنید خطایی
ممکن است رخ دهد. رویکردِ سلام شما را وادار می‌کند که \emph{آگاهانه} دربارهٔ
حالتِ خطا تصمیم بگیرید. این کمی بیشتر کد می‌خواهد، اما برنامه‌هایی می‌سازد که در
برابرِ شرایطِ ناخواسته مقاوم‌اند.
\end{warn}

\section{بررسیِ مرزها و اجرای امن}

یکی از رایج‌ترین خطاها، بیرون‌زدن از مرزِ آرایه است (که در فصلِ ۷ هشدارش را
دادیم). سلام می‌تواند در حالتِ \emph{امن} این خطاها را در زمانِ اجرا بگیرد. اگر
برنامه را با گزینهٔ \code{--safe} بسازید:

\begin{salamen}[language={}]
salam build app.salam --safe --output=app.exe
\end{salamen}

آنگاه هر دسترسیِ نامعتبر به آرایه، به‌جای رفتارِ نامشخص، برنامه را با یک پیامِ
روشن متوقف می‌کند. این برای یافتنِ خطاها در حینِ توسعه بسیار ارزشمند است.

\section{ترکیبِ تعلیق و مدیریتِ خطا}

این دو ابزار با هم می‌درخشند. الگوی رایج چنین است: منبع را بگیر، آزادسازی‌اش را
تعلیق کن، سپس کارها را انجام بده و در هر خطایی با خیالِ راحت \code{بازگشت} کن ---
چون می‌دانی \code{تعلیق} پاکسازی را تضمین کرده است:

\begin{salam}
تابع پردازش(داده صحیح۳۲) منطقی:
    تعلیق چاپ "منبع آزاد شد"
    اگر داده < 0:
        چاپ "ورودی نامعتبر"
        بازگشت نادرست
    پایان
    چاپ "پردازش:", داده
    بازگشت درست
پایان

تابع اصلی:
    پردازش(5)
    چاپ "---"
    پردازش(-1)
پایان
\end{salam}

در هر دو فراخوانی چه موفق و چه ناموفق پیامِ «منبع آزاد شد» چاپ می‌شود،
چون با \code{تعلیق} نشانه‌گذاری شده بود.

\section{خلاصهٔ فصل}

\begin{itemize}
  \item \code{تعلیق} دستوری را تا پایانِ تابع به تعویق می‌اندازد.
  \item هر مسیرِ خروج از تابع، \code{تعلیق}ها را اجرا می‌کند عالی برای
        آزادسازیِ منابع.
  \item چند \code{تعلیق} به ترتیبِ معکوس اجرا می‌شوند.
  \item سلام خطاها را \emph{صریح} مدیریت می‌کند: تابع‌ها نشانهٔ موفقیت
        برمی‌گردانند.
  \item گزینهٔ \code{--safe} خطاهای مرزِ آرایه را در زمانِ اجرا می‌گیرد.
  \item ترکیبِ \code{تعلیق} و مدیریتِ صریحِ خطا، کدِ مقاوم می‌سازد.
\end{itemize}

\section{تمرین‌ها}

\exercise تابعی بنویسید که در آغازش با \code{تعلیق} پیامی چاپ کند و سپس دو خطِ
دیگر. ترتیبِ خروجی را پیش‌بینی و سپس آزمایش کنید.

\exercise تابعی به نامِ \code{ریشهٔ\_دوم\_امن} بنویسید که اگر ورودی منفی بود،
\code{نادرست} برگرداند و نتیجه را از راهِ پارامترِ ارجاعی بدهد.

\exercise تابعی بنویسید که سه \code{تعلیق} داشته باشد و با اجرای آن، ترتیبِ
معکوس را تأیید کنید.

\exercise برنامه‌ای بنویسید که آرایه‌ای بسازد و عمداً به اندیسی بیرون از مرز
دسترسی پیدا کند؛ سپس آن را با \code{--safe} بسازید و رفتار را ببینید.

\exercise تابعی بنویسید که یک سن بگیرد و اگر بیرون از بازهٔ ۰ تا ۱۵۰ بود، با
\code{نادرست} خطا را اعلام کند.
