\chapter{برگهٔ مرجعِ سریع}%
\label{app:cheatsheet}

این پیوست رایج‌ترین الگوهای زبانِ سلام را در یک نگاه گرد آورده است برای
وقتی که فقط می‌خواهید شکلِ درستِ چیزی را سریع به یاد بیاورید.

\section{اسکلتِ برنامه}

\begin{salam}
تابع اصلی:
    چاپ "سلام، دنیا!"
تمام
\end{salam}

\section{متغیر و چاپ}

\begin{salam}
متغیر شمار: صحیح = 0      // تغییرپذیر
نام: رشته = "سارا"          // تغییرناپذیر
سن: خودکار = 30             // استنتاج
چاپ نام, سن               // چاپ با خطِ تازه
بنویس "بدونِ خطِ تازه "
\end{salam}

\section{عملگرها}

\begin{salam}
+  -  *  /  %  **          // حسابی
+=  -=  *=  /=  %=  **=     // انتساب ترکیبی
==  !=  <  >  <=  >=        // مقایسه
&&  ||  !                   // منطقی
الف + " " + ب              // پیوندِ رشته
\end{salam}

\section{شرط}

\begin{salam}
اگر سن >= 18:
    چاپ "بزرگسال"
وگرنه سن >= 13:
    چاپ "نوجوان"
وگرنه:
    چاپ "کودک"
تمام
\end{salam}

\section{حلقه‌ها}

\begin{salam}
تاوقتی شرط:  ...  تمام
تکرار 10 با i:  ...  تمام        // i از 0 تا 9
تکرار 5:  ...  تمام
تکرار 1 تا 10:  ...  تمام
تکرار 1 تا 10 با i:  ...  تمام   // بازهٔ شامل، با شمارنده
تکرار 0 تا 100 گام 10:  ...  تمام
// بشکن  /  ادامه
\end{salam}

\section{تابع}

\begin{salam}
تابع جمع(الف: صحیح, ب: صحیح): صحیح:
    بازگشت الف + ب
تمام
\end{salam}

\section{آرایه و بردار}

\begin{salam}
اعداد: صحیح[3] = [1, 2, 3]
چاپ اعداد[0], len(اعداد)

متغیر ب: وکتور<صحیح> = وکتور {}
ب.بیفزا(10)        // افزودن
ب.بگیر(0)[0]       // خواندن
ب.طول()            // طول
ب.دربیاور()        // برداشتنِ آخر
\end{salam}

\section{رشته}

\begin{salam}
س.طول()                 // طول
س.زیررشته(0, 3)         // برش
س.پیوست("!")            // چسباندن
س.بشکاف(",")            // شکستن به بردار
"42".به‌صحیح()          // به عدد
\end{salam}

\section{نگاشت}

\begin{salam}
متغیر م: نگاشت<رشته, صحیح> = نگاشت {}
م.درج("الف", 1)        // افزودن
م.بگیر("الف")          // خواندن
م.دارد("الف")          // وجود؟
م.اندازه()             // تعداد
\end{salam}

\section{ساختار و شمارشی}

\begin{salam}
ساختار نقطه:
    x: صحیح = 0
    y: صحیح = 0
    تابع جمع(): صحیح: بازگشت این.x + این.y تمام
تمام
متغیر پ: نقطه = نقطه { x = 3, y = 4 }
پ.جمع()

شمارش رنگ: قرمز, سبز, آبی تمام
رنگ.قرمز بعنوان صحیح
\end{salam}

\section{واسط و کلی‌سازی}

\begin{salam}
واسط شکل:
    تابع مساحت(): صحیح
تمام

تابع توصیف(ش: dyn شکل): چاپ ش.مساحت() تمام

تابع بیشینه<ت>(الف: ت, ب: ت): ت:
    اگر الف > ب: بازگشت الف تمام
    بازگشت ب
تمام
\end{salam}

\section{لامبدا و تعلیق}

\begin{salam}
دوبرابر: func(صحیح) صحیح = (ن: صحیح) => ن * 2
دوبرابر(21)

تعلیق چاپ "پاکسازی"     // در پایانِ تابع اجرا می‌شود
\end{salam}

\section{ماژول و خارجی}

\begin{salam}
واردکردن "ریاضی"
ریاضی.جذر(144.0)

خارجی:
    تابع printf(format: رشته, ...): صحیح
تمام
\end{salam}
