\chapter{برگهٔ مرجعِ سریع}
\label{app:cheatsheet}

این پیوست رایج‌ترین الگوهای زبانِ سلام را در یک نگاه گرد آورده است برای
وقتی که فقط می‌خواهید شکلِ درستِ چیزی را سریع به یاد بیاورید.

\section{اسکلتِ برنامه}

\begin{salam}
تابع اصلی:
    چاپ "سلام، دنیا!"
پایان
\end{salam}

\section{متغیر و چاپ}

\begin{salam}
متغیر شمار صحیح۳۲ = 0      // تغییرپذیر
نام رشته = "سارا"          // تغییرناپذیر
سن خودکار = 30             // استنتاج
چاپ نام, سن               // چاپ با خطِ تازه
بنویس "بدونِ خطِ تازه "
\end{salam}

\section{عملگرها}

\begin{salam}
+  -  *  /  %  **          // حسابی
==  !=  <  >  <=  >=        // مقایسه
&&  ||  !                   // منطقی
الف + " " + ب              // پیوندِ رشته
\end{salam}

\section{شرط}

\begin{salam}
اگر سن >= 18:
    چاپ "بزرگسال"
وگرنه سن >= 13:
    چاپ "نوجوان"
وگرنه:
    چاپ "کودک"
پایان
\end{salam}

\section{حلقه‌ها}

\begin{salam}
تاوقتی شرط:  ...  پایان
برای متغیر i = 0; i < 10; i = i + 1:  ...  پایان
تکرار 5:  ...  پایان
تکرار 1 تا 10:  ...  پایان
تکرار 0 تا 100 هر 10:  ...  پایان
// بشکن  /  ادامه
\end{salam}

\section{تابع}

\begin{salam}
تابع جمع(الف صحیح۳۲, ب صحیح۳۲) صحیح۳۲:
    بازگشت الف + ب
پایان
\end{salam}

\section{آرایه و بردار}

\begin{salam}
اعداد صحیح۳۲[3] = [1, 2, 3]
چاپ اعداد[0], len(اعداد)

متغیر ب وکتور<صحیح۳۲> = وکتور {}
ب.بیفزا(10)        // افزودن
ب.بگیر(0)[0]       // خواندن
ب.طول()            // طول
ب.دربیاور()        // برداشتنِ آخر
\end{salam}

\section{رشته}

\begin{salam}
س.طول()                 // طول
س.زیررشته(0, 3)         // برش
س.پیوست("!")            // چسباندن
س.بشکاف(",")            // شکستن به بردار
"42".به_صحیح()          // به عدد
\end{salam}

\section{نگاشت}

\begin{salam}
متغیر م نگاشت<رشته, صحیح۳۲> = نگاشت {}
م.درج("الف", 1)        // افزودن
م.بگیر("الف")          // خواندن
م.دارد("الف")          // وجود؟
م.اندازه()             // تعداد
\end{salam}

\section{ساختار و شمارشی}

\begin{salam}
ساختار نقطه:
    x صحیح۳۲ = 0
    y صحیح۳۲ = 0
    تابع جمع() صحیح۳۲: بازگشت این.x + این.y پایان
پایان
متغیر پ نقطه = نقطه { x = 3, y = 4 }
پ.جمع()

شمارش رنگ: قرمز, سبز, آبی پایان
رنگ.قرمز بعنوان صحیح۳۲
\end{salam}

\section{واسط و کلی‌سازی}

\begin{salam}
واسط شکل:
    تابع مساحت() صحیح۳۲
پایان

تابع توصیف(ش dyn شکل): چاپ ش.مساحت() پایان

تابع بیشینه<ت>(الف ت, ب ت) ت:
    اگر الف > ب: بازگشت الف پایان
    بازگشت ب
پایان
\end{salam}

\section{لامبدا و تعلیق}

\end{salam}
