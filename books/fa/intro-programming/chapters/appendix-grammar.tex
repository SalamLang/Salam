\chapter{دستورِ زبان (گرامر)}
\label{app:grammar}

این پیوست ساختارهای نحویِ اصلیِ زبانِ سلام را به‌صورتِ فشرده گرد آورده است. هدف،
مرجعی سریع برای یادآوریِ شکلِ درستِ هر ساختار است، نه تعریفِ صوریِ کامل.

\section{ساختارِ کلیِ برنامه}

یک برنامهٔ اجرایی از چند تعریف (تابع، ساختار، شمارشی، \ldots) ساخته می‌شود و
حتماً یک تابعِ \code{اصلی} دارد:

\begin{salam}
// تعریف‌های دلخواه (توابع، ساختارها، ...)
تابع اصلی:
    // بدنهٔ برنامه
پایان
\end{salam}

\section{اعلانِ متغیر}

\begin{salam}
متغیر نام نوع = مقدار      // تغییرپذیر
ثابت  نام نوع = مقدار      // تغییرناپذیر
نام نوع = مقدار            // تغییرناپذیر (شکلِ ساده)
نام خودکار = مقدار         // استنتاجِ نوع
\end{salam}

\section{تابع}

\begin{salam}
تابع نام(پارامتر۱ نوع, پارامتر۲ نوع) نوعِ‌خروجی:
    // بدنه
    بازگشت مقدار
پایان

تابع نام(پارامتر نوع):    // بدونِ مقدارِ بازگشتی
    // بدنه
پایان
\end{salam}

پارامترِ ارجاعی با \code{\&} مشخص می‌شود:

\begin{salam}
تابع نام(پارامتر &نوع):
    // تغییرِ پارامتر، متغیرِ بیرون را عوض می‌کند
پایان
\end{salam}

\section{شرط}

\begin{salam}
اگر شرط:
    // ...
وگرنه اگر شرطِ‌دیگر:
    // ...
وگرنه:
    // ...
پایان
\end{salam}

\section{حلقه‌ها}

\begin{salam}
تاوقتی شرط:
    // ...
پایان

برای متغیر i = آغاز; شرط; گام:
    // ...
پایان

تکرار N:              // N بار
تکرار آ تا ب:          // از آ تا بِ شامل
تکرار آ تا ب هر گ:     // با گامِ گ
\end{salam}

کنترلِ حلقه: \code{بشکن} (خروج) و \code{ادامه} (پرش به تکرارِ بعد).

\section{ساختار و شمارشی}

\begin{salam}
ساختار نام:
    میدان نوع = پیش‌فرض
    تابع متد() نوعِ‌خروجی:
        بازگشت این.میدان
    پایان
پایان

// ساختنِ نمونه:
نام { میدان = مقدار }

شمارش نام:
    عضو۱
    عضو۲ = 5
    عضو۳
پایان
// دسترسی: نام.عضو۱   و   نام.عضو۱ بعنوان صحیح۳۲
\end{salam}

\section{واسط و چندریختی}

\begin{salam}
واسط نام:
    تابع متد() نوعِ‌خروجی     // فقط امضا
پایان

تابع کاربر(پارامتر dyn نام):
    پارامتر.متد()
پایان
\end{salam}

\section{کلی‌سازی}

\begin{salam}
تابع نام<ت>(پارامتر ت) ت:
    // ...
پایان

ساختار نام<آ, ب>:
    میدان۱ آ
    میدان۲ ب
پایان
// کاربرد: نام<صحیح۳۲>   یا   نام<رشته, صحیح۳۲>
\end{salam}

\section{لامبدا و بستار}

\begin{salam}
// نوعِ تابع:
func(نوعِ‌ورودی) نوعِ‌خروجی

// لامبدای یک‌عبارتی:
(پارامتر نوع) => عبارت

// لامبدای چنددستوری:
() => :
    // دستورها
    بازگشت مقدار
پایان
\end{salam}

\section{ماژول و دسترسی}

\begin{salam}
بسته نام                       // در فایلِ کتابخانه
عمومی تابع ...                  // عضوِ عمومی
عمومی ثابت ...

واردکردن "نامِ‌ماژول"           // واردکردنِ ساده
واردکردن مستعار "فایل.salam"   // با نامِ مستعار
\end{salam}

\section{اجرای به‌تعویق‌افتاده و خارجی}

\begin{salam}
تعلیق دستور                    // در پایانِ تابع اجرا می‌شود

خارجی "C":
    تابع نامِ‌خارجی(پارامتر نوع) نوعِ‌خروجی
پایان

پیوند پویا "نامِ‌کتابخانه"      // پیوند به کتابخانهٔ خارجی
\end{salam}

\section{چیدمان (وب)}

\begin{salam}
صفحه:
    ویژگی = مقدار
    عنصر:
        ویژگی = مقدار
        عنصرِ‌تودرتو: ویژگی = مقدار پایان
    پایان
پایان
\end{salam}

\section{قاعده‌های کلی}

\begin{itemize}
  \item بلوک‌ها با \code{:} باز و با \code{پایان} بسته می‌شوند.
  \item سازنده‌های مقدار (ساختار، آرایه) از \code{\{\,\}} و \code{[\,]} استفاده
        می‌کنند، نه \code{پایان}.
  \item توضیحاتِ تک‌خطی با \code{//} آغاز می‌شوند.
  \item رشته‌ها میانِ \code{"\,"} و کاراکترها میانِ \code{'\,'} می‌آیند.
  \item ارقامِ محاسباتی همیشه لاتین‌اند (\code{2}، نه ۲).
  \item اندیسِ آرایه‌ها و بردارها از صفر آغاز می‌شود.
\end{itemize}
