\chapter{فهرستِ کاملِ واژگانِ کلیدی}
\label{app:keywords}

این پیوست مرجعِ کاملِ معادل‌های فارسی و انگلیسیِ زبانِ سلام است. هر واژه را
می‌توان به هر دو زبان نوشت و هر دو به یک برنامهٔ یکسان تبدیل می‌شوند.

\begin{note}
\code{\_} (یک زیرخط) شکلِ کوتاهِ \code{بنویس}/\code{print} و \code{\_\_} (دو
زیرخط) شکلِ کوتاهِ \code{چاپ}/\code{println} است. این میان‌برها در نمونه‌های
کوتاه به کار می‌روند.
\end{note}

\section{فرمان‌های کامپایلر}

\begin{center}
\begin{longtable}{l l}
\toprule
\textbf{فرمان} & \textbf{کار} \\
\midrule
\endhead
\code{salam exec app.salam} & اجرای بی‌درنگ با مفسر \\
\code{salam build app.salam --output=app.exe} & ساختِ فایلِ اجرایی \\
\code{salam build app.salam --safe} & ساخت با وارسیِ مرزِ آرایه \\
\code{salam build app.salam --keep-c} & نگه‌داشتنِ کدِ C تولیدشده \\
\code{salam obj app.salam} & ساختِ فایلِ شیء (.o) \\
\code{salam layout build page.salam} & ساختِ صفحهٔ وب (HTML/CSS/JS) \\
\code{salam layout build page.salam --inline} & صفحهٔ یکپارچه \\
\code{salam fmt app.salam} & قالب‌بندیِ کد \\
\code{salam fmt --check} & گزارشِ فایل‌های نامرتب \\
\code{salam debug app.salam} & ساخت و اجرای اشکال‌زدا \\
\code{salam memcheck app.salam} & وارسیِ نشتِ حافظه \\
\code{salam app.salam --emit-tokens-xml} & نمایشِ نشانه‌ها \\
\code{salam app.salam --emit-ast-xml} & نمایشِ درختِ نحوی \\
\code{salam -DFLAG build app.salam} & تعریفِ پرچمِ پیش‌پردازنده \\
\bottomrule
\end{longtable}
\end{center}
